//Extrayendo valores del xml

<raiz>
	<elemento1 id="1">
		<elemento2>aaa</elemento2>
	</elemento1>
	<elemento1 id="2">
		<elemento2>bbb</elemento2>
	</elemento1>
</tr>

Un valor discreto

<xsl:value-of select="raiz/elemento1[@id=2]/elemento2"/>

Una lista de valores

<xsl:for-each select="raiz/elemento1">
	<xsl:value-of select="elemento2"/>
	<xsl:value-of select="@id"/>
</xsl:for-each>

Ordenando la lista

<xsl:for-each select="raiz/elemento1">
	<xsl:sort select="elemento2" order="ascending/descending"/>

If

<xsl:if test="X &gt; 10">

Choose 

<xsl:choose>
  <xsl:when test="expression">
    ... 
  </xsl:when>
  <xsl:otherwise>
    ... 
  </xsl:otherwise>
</xsl:choose>

Variables:

<xsl:variable name="color">
	<xsl:choose>
		<xsl:when test="position() mod 2 = 0">
			#EEEEEE
		</xsl:when>
		<xsl:otherwise>
			#FFFFFF
		</xsl:otherwise>
	</xsl:choose>
</xsl:variable>

<tr bgcolor="{$color}">
